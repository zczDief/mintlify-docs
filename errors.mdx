def make_request_with_retry(
    url: str,
    headers: dict,
    max_retries: int = 3,
    base_delay: float = 1.0
) -> Optional[requests.Response]:
    """
    Faz requisi√ß√£o com retry e backoff exponencial
    """
    for attempt in range(max_retries + 1):
        try:
            response = requests.get(url, headers=headers, timeout=30)
            
            # Sucesso
            if response.status_code < 400:
                return response
            
            # N√£o fazer retry para erros do cliente (4xx)
            if 400 <= response.status_code < 500:
                if response.status_code == 429:  # Rate limit
                    # Aguardar tempo especificado no header
                    retry_after = response.headers.get('Retry-After')
                    if retry_after:

                        continue
                else:
                    # Outros erros 4xx n√£o devem ter retry
                    return response
            
            # Retry para erros 5xx
            if attempt < max_retries:
                delay = base_delay * (2 ** attempt) + random.uniform(0, 1)

                time.sleep(delay)

                return response

                print(f"Erro de conex√£o. Tentativa {attempt + 1}. Aguardando {delay:.2f}s...")

    return None

class JuditAPIClient {
    private $apiKey;
    private $baseHeaders;
    private $logger;
    
    public function __construct($apiKey) {
        $this->apiKey = $apiKey;
        $this->baseHeaders = ['api-key' => $apiKey];
        $this->logger = new class {
            public function info($message) {
                echo "[INFO] {$message}\n";
            }
            public function error($message) {
                echo "[ERROR] {$message}\n";
            }
        };
    }
    
    public function makeRequest($method, $url, $options = []) {
        // Faz requisi√ß√£o com tratamento de erro e retry
        
        $maxRetries = 3;
        $baseDelay = 1;
        
        for ($attempt = 0; $attempt <= $maxRetries; $attempt++) {
            $ch = curl_init();
            
            // Configurar headers
            $headers = array_merge($this->baseHeaders, $options['headers'] ?? []);
            $headerArray = [];
            foreach ($headers as $key => $value) {
                $headerArray[] = "{$key}: {$value}";
            }
            
            curl_setopt($ch, CURLOPT_URL, $url);
            curl_setopt($ch, CURLOPT_CUSTOMREQUEST, $method);
            curl_setopt($ch, CURLOPT_HTTPHEADER, $headerArray);
            curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
            curl_setopt($ch, CURLOPT_HEADER, true);
            curl_setopt($ch, CURLOPT_TIMEOUT, 30);
            
            if (isset($options['body'])) {
                curl_setopt($ch, CURLOPT_POSTFIELDS, $options['body']);
            }
            
            $response = curl_exec($ch);
            $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
            $headerSize = curl_getinfo($ch, CURLINFO_HEADER_SIZE);
            
            if ($response === false) {
                $error = curl_error($ch);
                curl_close($ch);
                
                $this->logger->error("Erro de conex√£o (tentativa " . ($attempt + 1) . "): {$error}");
                if ($attempt < $maxRetries) {
                    $delay = $baseDelay * (2 ** $attempt) + mt_rand(0, 1000) / 1000;
                    sleep((int)$delay);
                    continue;
                } else {
                    throw new Exception("Connection error: {$error}");
                }
            }
            
            $headers = substr($response, 0, $headerSize);
            $body = substr($response, $headerSize);
            
            // Log da requisi√ß√£o
            $this->logRequest($body, $httpCode, $attempt + 1);
            
            // Sucesso
            if ($httpCode < 400) {
                curl_close($ch);
                return ['status' => $httpCode, 'body' => $body, 'headers' => $headers];
            }
            
            // Tratamento de erros
            if (!$this->shouldRetry($httpCode, $attempt, $maxRetries)) {
                curl_close($ch);
                return ['status' => $httpCode, 'body' => $body, 'headers' => $headers];
            }
            
            // Aguardar antes do retry
            $delay = $this->calculateDelay($headers, $httpCode, $attempt, $baseDelay);
            sleep((int)$delay);
            
            curl_close($ch);
        }
        
        return null;
    }
    
    private function shouldRetry($httpCode, $attempt, $maxRetries) {
        // Determina se deve fazer retry
        
        if ($attempt >= $maxRetries) {
            return false;
        }
        
        // Retry para rate limit
        if ($httpCode === 429) {
            return true;
        }
        
        // Retry para erros 5xx
        if ($httpCode >= 500) {
            return true;
        }
        
        // N√£o fazer retry para outros erros
        return false;
    }
    
    private function calculateDelay($headers, $httpCode, $attempt, $baseDelay) {
        // Calcula delay para retry
        
        if ($httpCode === 429) {
            // Usar Retry-After se dispon√≠vel
            preg_match('/Retry-After: (\d+)/', $headers, $matches);
            if (isset($matches[1])) {
                return (int)$matches[1];
            }
        }
        
        // Backoff exponencial com jitter
        return $baseDelay * (2 ** $attempt) + mt_rand(0, 1000) / 1000;
    }
    
    private function logRequest($body, $httpCode, $attempt) {
        // Registra detalhes da requisi√ß√£o
        
        if ($httpCode >= 400) {
            try {
                $data = json_decode($body, true);
                $errorData = $data['error'] ?? [];

                    "Erro API (tentativa {$attempt}): " .
                    "Status {$httpCode}, " .
                    "C√≥digo: " . ($errorData['code'] ?? 'UNKNOWN') . ", " .
                    "Request ID: " . ($errorData['request_id'] ?? 'N/A')
                );
            } catch (Exception $e) {
                $this->logger->error(
                    "Erro API (tentativa {$attempt}): Status {$httpCode}"
                );
            }
        } else {
            $this->logger->info("Requisi√ß√£o bem-sucedida (tentativa {$attempt})");
        }
    }
