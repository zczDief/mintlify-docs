## Tratamento de Erros

### Callbacks e Hooks

```python

    print(f"ðŸš€ Iniciando {method} {url}")

    print(f"âœ… Sucesso: {response.status_code}")
```

## Exemplo Completo

```python
import os
import asyncio
from judit import JuditClient
from judit.types import RequestType, RequestStatus

async def main():
    # Inicializar cliente
    client = JuditClient(
        api_key=os.getenv('JUDIT_API_KEY'),
        enable_logging=True
    )
    
    try:
        # 1. Criar consulta
        print("ðŸ“‹ Criando consulta...")
        request = client.requests.create(
            request_type=RequestType.CPF,
            document="123.456.789-00",
            include_attachments=True
        )
        print(f"âœ… Consulta criada: {request.id}")
        
        # 2. Aguardar processamento
        print("â³ Aguardando processamento...")
        completed_request = client.requests.wait_for_completion(
            request_id=request.id,
            timeout=300
        )
        
        if completed_request.status == RequestStatus.COMPLETED:
            print("âœ… Consulta concluÃ­da!")
            
            # 3. Obter respostas
            responses = client.responses.list(
                request_id=request.id
            )
            
            print(f"ðŸ“„ {len(responses)} processos encontrados:")
            for response in responses:
                print(f"  - {response.process_number} ({response.court})")
                
                # 4. Criar tracking para processos ativos
                if response.status == "active":
                    tracking = client.tracking.create(
                        process_number=response.process_number,
                        webhook_url="https://seu-app.com/webhook"
                    )
                    print(f"  ðŸ‘ï¸ Tracking criado: {tracking.id}")
        
        else:
            print(f"âš ï¸ Consulta nÃ£o concluÃ­da: {completed_request.status}")
    
    except Exception as e:
        print(f"âŒ Erro: {e}")
    
    finally:
        # Fechar cliente
        client.close()

if __name__ == "__main__":
    asyncio.run(main())
```
