## Consultas (Requests)

### Criar Nova Consulta


## Tratamento de Erros

```python
from judit.exceptions import (
    JuditAPIError,
    AuthenticationError,
    ValidationError,
    RateLimitError,
    ResourceNotFoundError
)

try:
    request = client.requests.create(
        request_type=RequestType.CPF,
        document="cpf_invalido"
    )
    
except ValidationError as e:
    print(f"‚ùå Erro de valida√ß√£o: {e.message}")
    print(f"Campo: {e.field}")
    print(f"Valor fornecido: {e.provided_value}")
    
except AuthenticationError as e:
    print(f"üîê Erro de autentica√ß√£o: {e.message}")
    print("Verifique sua API Key")
    
except RateLimitError as e:
    print(f"‚è≥ Rate limit excedido: {e.message}")
    print(f"Aguarde {e.retry_after} segundos")
    
except ResourceNotFoundError as e:
    print(f"üîç Recurso n√£o encontrado: {e.message}")
    
except JuditAPIError as e:
    print(f"üö® Erro da API: {e.message}")
    print(f"C√≥digo: {e.error_code}")
    print(f"Request ID: {e.request_id}")
    
except Exception as e:
    print(f"üí• Erro inesperado: {e}")
```

### Callbacks e Hooks

```python
def on_request_start(method, url, **kwargs):
    print(f"üöÄ Iniciando {method} {url}")

def on_request_success(response):
    print(f"‚úÖ Sucesso: {response.status_code}")

def on_request_error(response):
    print(f"‚ùå Erro: {response.status_code}")

def on_retry(attempt, delay):
    print(f"üîÑ Tentativa {attempt}, aguardando {delay}s")

# Registrar callbacks
client.on('request_start', on_request_start)
client.on('request_success', on_request_success)
client.on('request_error', on_request_error)
client.on('retry', on_retry)
```

## Exemplo Completo

```python
import os
import asyncio
from judit import JuditClient
from judit.types import RequestType, RequestStatus

async def main():
    # Inicializar cliente
    client = JuditClient(
        api_key=os.getenv('JUDIT_API_KEY'),
        enable_logging=True
    )
    
    try:
        # 1. Criar consulta
        print("üìã Criando consulta...")
        request = client.requests.create(
            request_type=RequestType.CPF,
            document="123.456.789-00",
            include_attachments=True
        )
        print(f"‚úÖ Consulta criada: {request.id}")
        
        # 2. Aguardar processamento
        print("‚è≥ Aguardando processamento...")
        completed_request = client.requests.wait_for_completion(
            request_id=request.id,
            timeout=300
        )
        
        if completed_request.status == RequestStatus.COMPLETED:
            print("‚úÖ Consulta conclu√≠da!")
            
            # 3. Obter respostas
            responses = client.responses.list(
                request_id=request.id
            )
            
            print(f"üìÑ {len(responses)} processos encontrados:")
            for response in responses:
                print(f"  - {response.process_number} ({response.court})")
                
                # 4. Criar tracking para processos ativos
                if response.status == "active":
                    tracking = client.tracking.create(
                        process_number=response.process_number,
                        webhook_url="https://seu-app.com/webhook"
                    )
                    print(f"  üëÅÔ∏è Tracking criado: {tracking.id}")
        
        else:
            print(f"‚ö†Ô∏è Consulta n√£o conclu√≠da: {completed_request.status}")
    
    except Exception as e:
        print(f"‚ùå Erro: {e}")
    
    finally:
        # Fechar cliente
        client.close()

if __name__ == "__main__":
    asyncio.run(main())
```
